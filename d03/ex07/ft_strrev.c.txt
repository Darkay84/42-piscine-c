#include <stdio.h>

char	*ft_strrev(char *str)
{
    str = "asdf";
    int i;
    int j;
    char swp;
   
    j = 0;
    i = 0;
    while (str[i])
    {
        i++;
    }
    while (j++ < i--)
    {
        swp = str[j];
        str[j] = str[i];
        str[i] = swp;
    }
    return (str);
}
